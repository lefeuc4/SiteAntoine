function parseTitre(titre: string): { annee: string | null; nom: string } {
  const match = titre.match(/^(\d{4})\s*[---]-]\s*(.+)$/)
  if (match) return { annee: match[1], nom: match[2].trim() }
  return { annee: null, nom: titre }
}
